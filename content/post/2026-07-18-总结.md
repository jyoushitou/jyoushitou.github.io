#### 服务器端(Service)

```cpp
#include <boost/asio.hpp>
#include <iostream>
#include <set>
#include <memory>

using namespace boost::asio;
using namespace std;
const int max_length=1024;

typedef std::shared_ptr<ip::tcp::socket> socket_ptr;
std::set<std::shared_ptr<std::thread>>thread_set;

void seesion(socket_ptr sock)
{
    try
    {
        while(1)
        {
            char data[max_length];
            memset(data,'\0',max_length);
            boost::system::error_code error;
            //size_t length =boost::asio::read(sock,boost::asio::buffer(data,max_length),error);
            size_t length=sock->read_some(boost::asio::buffer(data,max_length),error);
            if(error==boost::asio::error::eof)
            {
                std::cout<<"connect closed by peer"<<std::endl;
                break;
            }
            else if(error)
            {
                throw boost::system::system_error(error);
            }
            cout<<"receivce from"<<sock->remote_endpoint().address().to_string()<<endl;
            cout<<"receivce message is"<<data<<endl;
            //回传数据
            boost::asio::write(*sock,boost::asio::buffer(data,length));
        }
    }
    catch(const std::exception& e)
    {
        std::cerr << e.what() << std::endl;
    }

}

void server(boost::asio::io_context& io_context,unsigned short port)
{
    ip::tcp::acceptor a(io_context,ip::tcp::endpoint(ip::tcp::v4(),port));
    while(1)
    {
        socket_ptr socket(new ip::tcp::socket(io_context));
        a.accept(*socket);
        auto t=std::make_shared<thread>(seesion,socket);
        thread_set.insert(t);
    }
}

int main()
{
    try
    {
        boost::asio::io_context ioc;
        server(ioc,10086);
        for(auto& t:thread_set)
        {
            t->join();
        }
    }
    catch(const std::exception& e)
    {
        std::cerr << e.what() << '\n';
    }

    return 0;
}
```

### 封装消息节点

```cpp
class MsgNode
{
public:
	MsgNode(const char* msg,int total_len):_total_len(total_len)//接收构造
	{
		_msg = new char[total_len];//为发送申请个缓冲区
		memcpy(_msg, msg, total_len);//拷贝数据
	}

	MsgNode(int total_len):_total_len(total_len),_cur_len(0)//发送构造
	{
		_msg = new char[total_len];//为发送的数据申请缓冲区
	}
	~MsgNode()
	{
		delete[] _msg;//删除创建的动态内存
	}
private:
	int _total_len;//总长度
	int _cur_len;//目前传入/传出长度
	char* _msg;//消息串
};
```

```cpp
// 使用 lambda 的版本
void Session::WriteToSocket(std::shared_ptr<SendNode> send_node) {
    auto self = shared_from_this();  // 确保 Session 对象存活


    this->_socket->async_write_some(
        boost::asio::buffer(send_node->_msg, send_node->_total_len),
        [this, self, send_node](const boost::system::error_code& error,
                                size_t bytes_transferred) {
            if (error) {
                std::cerr << "Write failed: " << error.message() << std::endl;
                Close();
                return;
            }

            std::cout << "Successfully wrote " << bytes_transferred
                      << " bytes out of " << send_node->_total_len << std::endl;

            if (bytes_transferred < send_node->_total_len) {
                send_node->_msg += bytes_transferred;
                send_node->_total_len -= bytes_transferred;
                WriteToSocket(send_node);  // 递归调用继续发送
            } else {
                std::cout << "Message sent completely" << std::endl;
            }
        }
    );
}
```

```cpp
void Session::WriteCallBackErr(const boost::system::error_code& ec, size_t bytes_transferred, shared_ptr<MsgNode> msg_node)
{
    // 判断是否还有未发送的数据
    if (bytes_transferred + msg_node->_cur_len < msg_node->_total_len)
    {
        // 更新已发送的字节数
        _send_node->_cur_len += bytes_transferred;

        // 继续发送剩余数据
        this->_socket->async_write_some(
            boost::asio::buffer(
                _send_node->_msg + _send_node->_cur_len,  // 剩余数据的起始位置
                _send_node->_total_len - _send_node->_cur_len  // 剩余数据长度
            ),
            boost::bind(&Session::WriteCallBackErr, this, _1, _2, _send_node)
        );
    }
}
```
