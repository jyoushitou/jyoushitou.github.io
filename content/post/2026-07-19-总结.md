#### 队列发送

##### 类结构

```cpp
class Session
{
public:
	Session(shared_ptr<tcp::socket> socket);
	void Connect(const tcp::endpoint& ep);
	void WriteCallBackErr(const boost::system::error_code& ec, size_t bytes_transferred,
		shared_ptr<MsgNode>);
	void WriteToSocketErr(const string buf);

	void WriteCallBack(const boost::system::error_code& ec, size_t bytes_transferred);
	void WriteToSocket(const string buf);
private:
	queue<shared_ptr<MsgNode>> _send_queue;
	shared_ptr<MsgNode> _send_node;
	bool _send_pending;
};
```

##### 目的

- 保证异步的时序性

##### WriteToSocket

```cpp
void Session::WriteToSocket(const string buf)
{
    _send_queue.emplace(new MsgNode(buf.c_str(), buf.length()));
    if (_send_pending)
    {
        return;
    }
    this->_socket->async_write_some(
        boost::asio::buffer(_send_node->_msg, _send_node->_total_len),
        boost::bind(&Session::WriteCallBack, this, _1, _2));
    _send_pending = 1;
}
```

##### WriteCallBack

```cpp
void Session::WriteCallBack(const boost::system::error_code& ec, size_t bytes_transferred)
{
    if (ec.value() != 0)
    {
        return;
    }

    auto& send_data = _send_queue.front();
    send_data->_cur_len += bytes_transferred;
    if (send_data->_cur_len < send_data->_total_len)
    {
        this->_socket->async_write_some(boost::asio::buffer(send_data->_msg +send_data->_cur_len,
            send_data->_total_len-send_data->_cur_len),
            boost::bind(&Session::WriteCallBack, this, _1, _2));
    }
    _send_queue.pop();
    if (_send_queue.empty())
    {
        _send_pending = 0;
    }
    if (!_send_queue.empty())
    {
        auto& send_data = _send_queue.front();
        send_data->_cur_len += bytes_transferred;
        if (send_data->_cur_len < send_data->_total_len)
        {
            this->_socket->async_write_some(boost::asio::buffer(send_data->_msg + send_data->_cur_len,
                send_data->_total_len - send_data->_cur_len),
                boost::bind(&Session::WriteCallBack, this, _1, _2));
        }
    }
}
```

```cpp
class Session
{
public:
    void ReadFromSocket();
    void ReadCallBack(const boost::system::error_code& ec, size_t bytes_transferred);

    shared_ptr<tcp::socket> _socket;
    shared_ptr<MsgNode> _recv_node;
    bool _recv_pending;
}
```

#### 分批读

```cpp
void Session::ReadFromSocket()
{
    if (_recv_pending)
    {
        return;
    }

    _recv_node = make_shared<MsgNode>(RECVSIZE);
    _socket->async_read_some(boost::asio::buffer(_recv_node->_msg, _recv_node->_total_len),
        boost::bind(&Session::ReadCallBack, this, _1, _2));
    _recv_pending = 1;
}
//回调函数
void Session::ReadCallBack(const boost::system::error_code& ec, size_t bytes_transferred)
{
    _recv_node->_cur_len += bytes_transferred;
    if (_recv_node->_cur_len < _recv_node->_total_len)
    {
        this->_socket->async_write_some(boost::asio::buffer(_recv_node->_msg + _recv_node->_cur_len,
            _recv_node->_total_len - _recv_node->_cur_len),
            boost::bind(&Session::ReadCallBack, this, _1, _2));
        return;
    }

    _recv_pending = 0;
    _recv_node = nullptr;
}
```

#### 一次性读

```cpp
void Session::ReadAllFromSocket()
{
    if (_recv_pending)
    {
        return;
    }

    _recv_node = make_shared<MsgNode>(RECVSIZE);
    _socket->async_receive(boost::asio::buffer(_recv_node->_msg, _recv_node->_total_len),
        boost::bind(&Session::ReadAllCallBack, this, _1, _2));
    _recv_pending = 1;
}
//回调函数
void Session::ReadAllCallBack(const boost::system::error_code& ec, size_t bytes_transferred)
{
    _recv_node->_cur_len += bytes_transferred;
    _recv_node = nullptr;
    _recv_pending = 0;
}
```

## 约束

### primary key 主键

- 唯一标识每一行，非空且唯一

### unique 唯一约束

- 保证列的值不重复（允许NULL）

### NOT NULL 保证列的值不能为空

- 设置插入数据不能为空

### DEFAULT 默认值

- 未指定值时使用默认值

### foreign key 外键约束

- 保证表与表之间的数据一致性

### check 检查约束

- 检查列的值是否满足某个条件

### 3.7join多表关联查询
#### 3.7.1INNER JOIN内连接

- 通过ON去区分，此为当s中的class_id=c的id相同时，合成一张表显示

- 左表全部，右表匹配不到为 NULL
#### 3.7.3RIGHT JOIN
```sql
SELECT s.name,c.* FROM student s RIGHT JOIN class c ON s.class_id = c.id;
```
- 右表全部，左表匹配不到为 NULL
#### 3.7.4全连接
- MySQL 不支持 FULL JOIN，用 UNION 模拟
#### 3.7.5CROSS JOIN交叉连接
```sql
SELECT * FROM student CROSS JOIN class;
```
- 笛卡尔积，慎用
- 将两张表的全部可能列出
### 3.8子查询
#### 3.8.1标量子查询
```sql

```
- 查询班级名称为"一班"的所有学生的姓名。
- 先查询小括号里的查询
#### 3.8.2IN子查询
```sql
SELECT * FROM student WHERE class_id IN (SELECT id FROM class WHERE grade=3);
```
- 查询所有年级为 3 的班级中的学生信息。
- 先查询小括号里的查询
#### 3.8.1EXISTS子查询
```sql
SELECT * FROM student s WHERE EXISTS (SELECT 1 FROM class c WHERE c.id=s.class_id);
```
- 查询所有存在对应班级的学生信息。
- 检查 class 表中是否存在与当前学生班级编号匹配的记录
### 3.9UNION 合并查询结果
#### 3.9.1UNION 自动去重
```sql
SELECT name FROM student WHERE sex='男' UNION SELECT name FROM teacher;
```
- 将学生表中所有男生的姓名和教师表中所有教师的姓名合并到一个结果集中，并自动去除重复的姓名。
#### 3.9.2UNION ALL 不去重，效率更高
```sql
SELECT name FROM student UNION ALL SELECT name FROM teacher;
```

- 每次执行用户操作后需要刷新

#### 4.7.1收回查询和插入权限
```sql
REVOKE SELECT,INSERT ON test_db.* FROM 'user1'@'localhost';
```
#### 4.7.2回收全部权限
```sql
REVOKE ALL PRIVILEGES, GRANT OPTION FROM 'user1'@'localhost';
```
### 4.8查看用户权限
```sql
SHOW GRANTS FOR 'user1'@'localhost';
```
## 5.TCL
### 5.1开启事物
```sql
START TRANSACTION;
BEGIN;
```
### 5.2提交事务
```sql
COMMIT;
```
- 永久生效
### 5.3回滚事务
```sql
ROLLBACK;
```
- 撤销未提交操作
### 5.4保存点局部回滚
```sql
SAVEPOINT sp1;
ROLLBACK TO sp1;
```
- SAVEPOINT sp1;:标记回滚点
- ROLLBACK TO sp1;:回滚到sp1点
