### 异步Demo
#### Session
- TCP 会话（Session）
- 封装 TCP 套接字
- 异步读写框架
- 数据缓冲区
##### 源码
###### 头文件
```cpp
class Session
{
public:
	Session(io_context& ioc):_socket(ioc){}//构造函数
	tcp::socket& Socket() {//返回套接字
		return _socket;
	}

	void Start();//启动会话
private:
	void handle_read(const boost_error_code& error,size_t bytes_transferred );//异步读回调函数
	void handle_write(const boost_error_code& error);//异步写回调函数
	tcp::socket _socket;//套接字
	enum {max_length=1024};//缓冲区最大长度
	char _data[max_length];//缓冲区
};
```
###### 实现
```cpp
void Session::Start()
{
	memset(_data, 0, max_length);//初始化缓冲区
	_socket.async_read_some(boost::asio::buffer(_data, max_length),
		bind(&Session::handle_read, this, std::placeholders::_1, std::placeholders::_2));//异步读
}
void Session::handle_read(const boost_error_code& error, size_t bytes_transferred)//异步读回调函数
{
	if (!error)
	{
		cout << "server receive data is" << _data << endl;
		boost::asio::async_write(_socket, boost::asio::buffer(_data, bytes_transferred),
			std::bind(&Session::handle_write, this, std::placeholders::_1));	//异步写
	}
	else
	{
		cout << "have error in read,error code is" << error.value() << endl;
		delete this;
	}
}
void Session::handle_write(const boost_error_code& error)//异步写回调函数
{
	if (!error)
	{
		memset(_data, 0, max_length);
		_socket.async_read_some(boost::asio::buffer(_data, max_length),
			bind(&Session::handle_read, this, std::placeholders::_1, std::placeholders::_2));	//异步读
		//也可以使用Start()

	}
	else
	{
		cout << "have error in write,error code is" << error.value() << endl;
		delete this;
	}
}
```
#### Server
##### 源码
###### 头文件
```cpp
class Server
{
public:
	Server(io_context& ioc, short port);
private:
	void start_accept();
	void handle_accept(Session* new_session,const boost_error_code& error);
	io_context& _ioc;
	tcp::acceptor _acceptor;
};
```
##### 实现
```cpp
Server::Server(io_context& ioc, short port) :_ioc(ioc), _acceptor(ioc, tcp::endpoint(tcp::v4(), port))
{
	cout << "Server starting" << endl;
	start_accept();
}

void Server::start_accept()
{
	Session* new_session = new Session(_ioc);
	_acceptor.async_accept(new_session->Socket(), std::bind(&Server::handle_accept, this, new_session,
		std::placeholders::_1));
}

void Server::handle_accept(Session* new_session, const boost_error_code& error)
{
	if (!error)
	{
		new_session->Start();
	}
	else {
		delete new_session;
	}
	start_accept();
}
```

