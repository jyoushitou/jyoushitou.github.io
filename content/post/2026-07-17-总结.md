void write_to_socket(asio::ip::tcp::socket& sock)
{
	std::string buf = "hello world";
	std::size_t total_bytes_written = 0;
	//循环发送
	while (total_bytes_written!=buf.length())
	{
		total_bytes_written += sock.write_some(asio::buffer(buf.c_str()+total_bytes_written,
			buf.length() - total_bytes_written));//循环的发送，一次发送最大的数据，并累加到total_bytes_written
	}
}

int send_data_by_write_some()
{
	std::string raw_ip_address = "192.168.0.60";//地址
	unsigned short port_num =3333;//端口
	try
	{
		asio::ip::tcp::endpoint ep(asio::ip::make_address(raw_ip_address), port_num);//这段代码是使用 Boost.Asio（或独立 Asio 库）创建一个 TCP 端点（endpoint），用于表示一个网络通信的目标地址和端口。
		//asio::io_service ios;
		asio::io_context ioc;//TCP上下文
		asio::ip::tcp::socket sock(ioc, ep.protocol());//创建socket
		sock.connect(ep);//socket连接
		write_to_socket(sock);//发送数据
	}
	catch (system::system_error& e)
	{
		std::cout << e.what() << std::endl;
		return e.code().value();
	}
	return 0;
}

int send_data_by_write_some()
{

	unsigned short port_num = 3333;
	try
	{
		asio::ip::tcp::endpoint ep(asio::ip::make_address(raw_ip_address), port_num);

		if (send_length <= 0)
		{
			return 0;
		}
	}
	catch (system::system_error& e)
	{
		std::cout << e.what() << std::endl;
		return e.code().value();
	}
	return 0;
}

//循环读
std::string read_from_socket(asio::ip::tcp::socket& sock)
{
	const unsigned char MESSAGE_SIZE = 7;
	char buf[MESSAGE_SIZE];
	std::size_t total_bytes_read = 0;
	while (total_bytes_read != MESSAGE_SIZE)
	{
		total_bytes_read += sock.read_some(asio::buffer
		(buf + total_bytes_read, MESSAGE_SIZE - total_bytes_read));
	}
	return std::string(buf, total_bytes_read);
}
