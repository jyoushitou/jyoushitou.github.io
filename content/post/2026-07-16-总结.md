### 解析域名

```cpp
int dns_conect_to_end()
{
	std::string host = "example.com";//域名
	std::string port_num = "3333";//端口
	asio::io_context ios;//上下文
	asio::ip::tcp::resolver resolver(ios);//创建dns解析器对象
	boost::system::error_code ec;
	auto results = resolver.resolve(host, port_num, ec);//解析域名
	if (ec)
	{
		std::cout << "DNS 解析失败，错误码 = " << ec.value() << "，信息：" << ec.message() << std::endl;
		return ec.value();
	}
	return 0;
}
```

### 服务器端的连接代码
```cpp
int acceptor_new_connection()
{
	const int BACKLOG_SIZE = 30;//等待队列的最大长度
	unsigned short port_num = 3333;
	asio::ip::tcp::endpoint ep(asio::ip::address_v4::any(),port_num);//指定所有的网卡为接收网卡
	asio::io_context ios;//TCP上下文
	try
	{
		asio::ip::tcp::acceptor acceptor(ios, ep.protocol());//TCP协议
		acceptor.bind(ep);//绑定网卡
		acceptor.listen(BACKLOG_SIZE);//监听队列
		asio::ip::tcp::socket sock(ios);//创建socket
		acceptor.accept(sock);//等待连接
	}
	catch (system::system_error& e)
	{
		std::cout << e.what() << std::endl;
	}
}
```

### 创建buffer
- 用于发送数据的对象
```cpp
//只读字符串
void use_buffer_str()
{
	std::string buf = "hello world";
	asio::const_buffer out(buf.c_str(), buf.length());
}
//典型用途：发送字符串数据到网络连接。

//可读写数组的
void use_buffer_array()
{
	const size_t BUF_SIZE_BYTES = 20;
	std::unique_ptr<char[]>buf(new char[BUF_SIZE_BYTES]);
	auto input_buf = asio::buffer(static_cast<void*>(buf.get()),BUF_SIZE_BYTES);
}
//典型用途：准备缓冲区来接收从网络连接读取的数据。
```
- 这两个函数本质上是 Asio 中发送和接收数据时创建 buffer 的标准模式。
