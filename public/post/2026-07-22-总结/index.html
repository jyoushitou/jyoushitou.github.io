class Session:public enable_shared_from_this<Session>
{
public:
	Session(io_context& ioc,Server* server):_socket(ioc),_server(server)
	{
		boost::uuids::uuid a_uuid = boost::uuids::random_generator()();
		uuid = boost::uuids::to_string(a_uuid);
	}
	tcp::socket& Socket() {
		return _socket;
	}

	string& Get_uuid() { return uuid; }

	void Send(char* msg, int max_length);

	void Start();
private:
	void handle_read(const boost_error_code& error, size_t bytes_transferred);
	void handle_write(const boost_error_code& error, size_t bytes_transferred);
	tcp::socket _socket;
	enum {max_length=1024};
	char _data[max_length];
	Server* _server;
	string uuid;
	
	queue<shared_ptr<MsgNode>>_send_que;
	mutex _send_lock;

};

Session.cpp 
#include "Session.h"
#include <iostream>
using namespace std;
using namespace boost::asio;

void Session::Send(char* msg, int max_length)
{
	auto node = make_shared<MsgNode>(msg, max_length);
	{
		lock_guard<mutex> lock(_send_lock);
		_send_que.push(node);
		// 如果已有正在进行的发送，则直接入队等待
		if (_send_que.size() > 1) return;
	}
	// 启动异步写；使用 lambda 捕获 shared_ptr 保证对象存活
	auto self = shared_from_this();
	async_write(_socket, buffer(node->_data, node->_max_len),
		[self](const boost_error_code& ec, size_t bytes_transferred) {
			self->handle_write(ec, bytes_transferred);
		});
}

void Session::Start()
{
	memset(_data, 0, max_length);
	auto self = shared_from_this();
	_socket.async_read_some(boost::asio::buffer(_data, max_length),
		[self](const boost_error_code& ec, size_t bytes_transferred) {
			self->handle_read(ec, bytes_transferred);
		});
}

void Session::handle_read(const boost_error_code& error, size_t bytes_transferred)
{
	if (!error)
	{
		// 注意数据可能包含二进制，使用 write 指定长度输出
		cout << "server receive data is ";
		cout.write(_data, bytes_transferred);
		cout << endl;

		// 发送收到的数据（Send 会复制数据并入队）
		Send(_data, static_cast<int>(bytes_transferred));

		// 准备下一次读取
		memset(_data, 0, max_length);
		auto self = shared_from_this();
		_socket.async_read_some(boost::asio::buffer(_data, max_length),
			[self](const boost_error_code& ec, size_t bt) {
				self->handle_read(ec, bt);
			});
	}
	else
	{
		cout << "have error in read,error code is " << error.value() << endl;
		_server->ClearSession(uuid);
	}
}

void Session::handle_write(const boost_error_code& error, size_t bytes_transferred)
{
	if (!error)
	{
		shared_ptr<MsgNode> next;
		{
			lock_guard<mutex>lock(_send_lock);
			if (!_send_que.empty())
			{
				_send_que.pop();
			}
			if (!_send_que.empty())
			{
				next = _send_que.front();
			}
		}
		if (next)
		{
			auto self = shared_from_this();
			async_write(_socket, buffer(next->_data, next->_max_len),
				[self](const boost_error_code& ec, size_t bt) {
					self->handle_write(ec, bt);
				});
		}
	}
	else
	{
		cout << "have error in write,error code is " << error.value() << endl;
		_server->ClearSession(uuid);
	}
}

Server::Server(io_context& ioc, short port) :_ioc(ioc), _acceptor(ioc, tcp::endpoint(tcp::v4(), port))
{
	cout << "Server starting" << endl;
	start_accept();
}

void Server::start_accept()
{

	shared_ptr<Session> new_session=make_shared<Session>(_ioc,this);
	_acceptor.async_accept(new_session->Socket(), std::bind(&Server::handle_accept, this, new_session,
		std::placeholders::_1));//加入this使其变为普通函数对象
	//shared_ptr不会随右括号而关闭
}

void Server::handle_accept(shared_ptr<Session> new_session, const boost_error_code& error)
{
	if (!error)
	{
		new_session->Start();
		_session.insert(make_pair(new_session->Get_uuid(), new_session));
	}
	else {
	}
	start_accept();
}

void Server::ClearSession(string uuid)
{
	_session.erase(uuid);
}
