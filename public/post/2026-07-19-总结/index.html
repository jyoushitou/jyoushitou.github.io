void Session::WriteToSocket(const string buf)
{
    _send_queue.emplace(new MsgNode(buf.c_str(), buf.length()));
    if (_send_pending)
    {
        return;
    }
    this->_socket->async_write_some(
        boost::asio::buffer(_send_node->_msg, _send_node->_total_len),
        boost::bind(&Session::WriteCallBack, this, _1, _2));
    _send_pending = 1;
}
WriteCallBack 
void Session::WriteCallBack(const boost::system::error_code& ec, size_t bytes_transferred)
{
    if (ec.value() != 0)
    {
        return;
    }

    auto& send_data = _send_queue.front();
    send_data->_cur_len += bytes_transferred;
    if (send_data->_cur_len < send_data->_total_len)
    {
        this->_socket->async_write_some(boost::asio::buffer(send_data->_msg +send_data->_cur_len,
            send_data->_total_len-send_data->_cur_len),
            boost::bind(&Session::WriteCallBack, this, _1, _2));
    }
    _send_queue.pop();
    if (_send_queue.empty())
    {
        _send_pending = 0;
    }
    if (!_send_queue.empty())
    {
        auto& send_data = _send_queue.front();
        send_data->_cur_len += bytes_transferred;
        if (send_data->_cur_len < send_data->_total_len)
        {
            this->_socket->async_write_some(boost::asio::buffer(send_data->_msg + send_data->_cur_len,
                send_data->_total_len - send_data->_cur_len),
                boost::bind(&Session::WriteCallBack, this, _1, _2));
        }
    }
}

void Session::ReadCallBack(const boost::system::error_code& ec, size_t bytes_transferred)
{
    _recv_node->_cur_len += bytes_transferred;
    if (_recv_node->_cur_len < _recv_node->_total_len)
    {
        this->_socket->async_write_some(boost::asio::buffer(_recv_node->_msg + _recv_node->_cur_len,
            _recv_node->_total_len - _recv_node->_cur_len),
            boost::bind(&Session::ReadCallBack, this, _1, _2));

    _recv_pending = 0;
    _recv_node = nullptr;

约束

primary key 主键

  • 唯一标识每一行，非空且唯一

unique 唯一约束

  • 保证列的值不重复（允许NULL）

NOT NULL 保证列的值不能为空

  • 设置插入数据不能为空

DEFAULT 默认值

  • 未指定值时使用默认值

foreign key 外键约束

  • 保证表与表之间的数据一致性

check 检查约束

  • 检查列的值是否满足某个条件

1.DDL

1.1数据库

1.1.1查看所有数据库

1.1.2创建数据库 

-- 基础创建
CREATE DATABASE db_name;

-- 不存在才创建，指定字符集、排序规则
CREATE DATABASE IF NOT EXISTS test_db DEFAULT CHARACTER SET utf8mb4 DEFAULT COLLATE utf8mb4_unicode_ci;
  • 这个意思为创建test_db的数据库，并使其保存格式默认为utf8mb4，默认排序方式为utf8mb4的排序法,IF后面的内容可省略

SHOW CREATE DATABASE test_db;
  • 可以便于迁移数据库

1.1.6删除数据库 

DROP DATABASE IF EXISTS test_db;
  • IF EXISTS可以省略，检查是否有这个名字的数据库

3.DQL

3.1基础查询

3.1.1查询所有数据 

selsect * from students;

3.1.2指定列、别名 

SELECT id AS 编号, name 姓名 FROM student;

3.1.3去重 DISTINCT 

SELECT DISTINCT class_id FROM student;
  • 从 student 表中查询所有不重复的 class_id 值。

3.1.4四则运算、函数 

SELECT id, age+1, CONCAT(name,'-',sex) FROM student;
  • 从 student 表中查询 id、age+1 的计算结果，以及将 name 和 sex 拼接后的新字符串。

3.1.5常量列 

SELECT id, '学生' AS type FROM student;
  • 从 student 表中查询 id 列，并为每一行添加一个固定值为 ‘学生’ 的 type 列。
  • type为列名

3.2WHERE 条件过滤

3.2.1运算符

  • = != > >= < <= AND OR NOT LIKE IN BETWEEN IS NULL

3.2.2精准搜索 

SELECT * FROM student
WHERE age >= 18 AND sex='男';

3.2.3模糊搜索 

SELECT * FROM student WHERE name LIKE '张%';
SELECT * FROM student WHERE name LIKE '%三';
SELECT * FROM student WHERE name LIKE '_明';

3.2.4区间搜索 

SELECT * FROM student WHERE age BETWEEN 18 AND 22;
  • BETWEEN 18 AND 22表示18-22的区间

3.2.5枚举匹配 

SELECT * FROM student WHERE class_id IN (1,3,5);

3.2.6空值判断 

SELECT * FROM student WHERE phone IS NULL;
SELECT * FROM student WHERE phone IS NOT NULL;
  • 不能用=NULL

3.3COUNT SUM AVG MAX MIN聚合函数 

SELECT
    COUNT(*) AS 总人数,
    COUNT(id) AS 有效id数,
    SUM(age) AS 年龄总和,
    AVG(age) AS 平均年龄,
    MAX(age) AS 最大年龄,
    MIN(age) AS 最小年龄
FROM student;
  • 将多行数据聚合成一行结果
  • 通常与 GROUP BY 配合使用，按组统计
  • 单独使用时，对整个表进行统计

3.4GROUP BY 分组 + HAVING 分组过滤

3.4.1分组统计 

SELECT class_id, COUNT(*) AS num FROM student GROUP BY class_id
  • 按 class_id 分组，统计每个班级的学生人数。

3.4.2过滤 

HAVING COUNT(*) >= 10;
  • HAVING不能用WHERE替代

3.4.3多字段分组 

SELECT class_id, sex, AVG(age) FROM student GROUP BY class_id, sex;
  • 按班级和性别进行分组，统计每个班级中不同性别的平均年龄。

3.5ORDER BY 排序

  • ASC升序，DESC为降序

3.5.1单字段排序 

SELECT * FROM student ORDER BY age DESC;

3.5.2多字段排序 

SELECT * FROM student ORDER BY class_id ASC, age DESC;

3.6LIMIT 分页 

-- 取前10条
SELECT * FROM student LIMIT 10;
-- 偏移5，取10条（第6~15行）
SELECT * FROM student LIMIT 5,10;

3.7join多表关联查询

3.7.1INNER JOIN内连接 

SELECT s.id,s.name,c.class_name FROM student s INNER JOIN class c ON s.class_id = c.id;
  • 通过ON去区分，此为当s中的class_id=c的id相同时，合成一张表显示
  • 只匹配两边都存在数据

3.7.2LEFT JOIN 

SELECT s.*, c.class_name FROM student s LEFT JOIN class c ON s.class_id = c.id;
  • 左表全部，右表匹配不到为 NULL

3.7.3RIGHT JOIN 

SELECT s.name,c.* FROM student s RIGHT JOIN class c ON s.class_id = c.id;
  • 右表全部，左表匹配不到为 NULL

3.7.4全连接

  • MySQL 不支持 FULL JOIN，用 UNION 模拟

3.7.5CROSS JOIN交叉连接 

SELECT * FROM student CROSS JOIN class;
  • 笛卡尔积，慎用
  • 将两张表的全部可能列出

3.8子查询

3.8.1标量子查询 

SELECT name FROM student WHERE class_id = (SELECT id FROM class WHERE class_name='一班');
  • 查询班级名称为"一班"的所有学生的姓名。
  • 先查询小括号里的查询

3.8.2IN子查询 

SELECT * FROM student WHERE class_id IN (SELECT id FROM class WHERE grade=3);
  • 查询所有年级为 3 的班级中的学生信息。
  • 先查询小括号里的查询

3.8.1EXISTS子查询 

SELECT * FROM student s WHERE EXISTS (SELECT 1 FROM class c WHERE c.id=s.class_id);
  • 查询所有存在对应班级的学生信息。
  • 检查 class 表中是否存在与当前学生班级编号匹配的记录

3.9UNION 合并查询结果

3.9.1UNION 自动去重 

SELECT name FROM student WHERE sex='男' UNION SELECT name FROM teacher;
  • 将学生表中所有男生的姓名和教师表中所有教师的姓名合并到一个结果集中，并自动去除重复的姓名。

3.9.2UNION ALL 不去重，效率更高 

SELECT name FROM student UNION ALL SELECT name FROM teacher;

4.DCL

4.1创建用户

4.1.1用户名@访问主机 

CREATE USER 'user1'@'localhost' IDENTIFIED BY '123456';
  • 只允许本机访问
  • IDENTIFIED BY设置密码

4.1.2任意主机访问% 

CREATE USER 'user2'@'%' IDENTIFIED BY '654321';

4.2修改密码 

ALTER USER 'user1'@'localhost' IDENTIFIED BY 'newpwd';
SET PASSWORD FOR 'user1'@'localhost' = PASSWORD('newpwd');
  • 可以使用IDENTIFIED BY去修改，或者直接赋值

4.3删除用户 

DROP USER IF EXISTS 'user1'@'localhost';

4.4查看所有用户 

SELECT user,host FROM mysql.user;

4.5GRANT 授予权限 

GRANT 权限列表 ON 库.表 TO '用户'@'主机' [WITH GRANT OPTION];
  • 常用权限：SELECT INSERT UPDATE DELETE CREATE DROP ALTER INDEX REFERENCES ALL PRIVILEGES

4.5.1给user1授予test_db所有表查询、插入权限 

GRANT SELECT,INSERT ON test_db.* TO 'user1'@'localhost';
  • 将test_db的所有表的查询和插入权限给user1

4.5.2授予单表权限 

GRANT SELECT,UPDATE ON test_db.student TO 'user1'@'localhost';

4.5.3全部权限 

GRANT ALL PRIVILEGES ON test_db.* TO 'user1'@'localhost';
  • 此数据库的所有权限赋予用户user1
  • 不能管理用户

4.5.4授权管理员权限 

GRANT ALL PRIVILEGES ON *.* TO 'root'@'%' WITH GRANT OPTION;
  • 管理员权限，可以管理用户，拥有与root等价的权限

4.6刷新 

FLUSH PRIVILEGES;
  • 每次执行用户操作后需要刷新

4.7回收权限

4.7.1收回查询和插入权限 

REVOKE SELECT,INSERT ON test_db.* FROM 'user1'@'localhost';

4.7.2回收全部权限 

REVOKE ALL PRIVILEGES, GRANT OPTION FROM 'user1'@'localhost';

4.8查看用户权限 

SHOW GRANTS FOR 'user1'@'localhost';

5.1开启事物
